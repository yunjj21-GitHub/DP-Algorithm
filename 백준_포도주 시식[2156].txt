// 백준 2156 : 포도주 시식
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int n;
vector<int> v(10001); // 포도주 정보를 저장
int dp[10001];

int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> v[i];
	}

	dp[1] = v[1];
	dp[2] = v[1] + v[2];
	dp[3] = max(v[2] + v[1], v[3] + v[2] + dp[0]);
	dp[3] = max(dp[3], v[3] + dp[1]);
	for (int i = 4; i <= n; i++) {
		dp[i] = max(v[i - 1] + v[i - 2] + dp[i - 4], v[i] + v[i - 1] + dp[i - 3]);
		dp[i] = max(dp[i], v[i] + dp[i - 2]);
	}

	int ans = 0;
	for (int i = 1; i <= n; i++) {
		if (dp[i] > ans) ans = dp[i];
	}
	cout << ans << '\n';
}
