#include <vector>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;

int main() {
	int T;
	cin >> T;
	vector<int> answer;
	while (T--) {
		int n;
		cin >> n;
		vector<vector<int>> score(2, vector<int>(n));
		vector<vector<int>> dp(2, vector<int>(n));
		for (int i = 0; i < 2; i++) {
			for (int j = 0; j < n; j++) {
				cin >> score[i][j];
			}
		}

		dp[0][0] = score[0][0];
		dp[1][0] = score[1][0];
		for (int i = 1; i < n; i++) {
			for (int j = 0; j < 2; j++) {
				dp[j][i] = max(score[j][i] + dp[1 - j][i - 1], dp[j][i - 1]);
			}
		}

		answer.push_back(max(dp[0][n - 1], dp[1][n - 1]));
	}

	for (int ans : answer) {
		cout << ans << '\n';
	}
}
