// 백준 10844 : 쉬운 계단 수
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int DIVIDER = 1000000000;
int n;
int dp[101][10]; // dp[자릿수(1~100)][숫자(0~9)]
long long ans = 0;


int main() {
	cin >> n;

	for (int i = 1; i < 10; i++) {
		dp[1][i] = 1;
	}

	for (int i = 2; i <= n; i++) { // 자릿수 (2~n)
		for (int j = 0; j < 10; j++) { // 숫자 (0~9)
			if (j == 0) {
				dp[i][0] = dp[i - 1][j + 1] % DIVIDER;
			}
			else if (j == 9) {
				dp[i][9] = dp[i - 1][j - 1] % DIVIDER;
			}
			else {
				dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j + 1] % DIVIDER;
			}
		}
	}

	// 답안 구하기
	for (int i = 0; i < 10; i++) {
		ans += dp[n][i];
	}
	ans %= DIVIDER;
	cout << ans << '\n';
}
