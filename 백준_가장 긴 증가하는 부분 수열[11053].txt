// 백준 11053 : 가장 긴 증가하는 부분 수열
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int n;
vector<int> num(1000);

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> num[i];
	}

	int* dp = new int[n];
	for (int i = 0; i < n; i++) {
		int maximum = 0;
		for (int j = 0; j < i; j++) {
			if (num[j] < num[i]) {
				maximum = maximum < dp[j] ? dp[j] : maximum;
			}
		}
		dp[i] = maximum + 1;
	}

	int ans = 0;
	for (int i = 0; i < n; i++) {
		if (ans < dp[i]) ans = dp[i];
	}
	cout << ans << '\n';
}
