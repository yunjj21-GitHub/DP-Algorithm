#include <vector>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

int N;
int T[1600000];
int P[1600000];
int dp[1600000];

int main() {
	cin >> N;
	for (int i = 1; i <= N; i++) {
		cin >> T[i] >> P[i];
	}

	for (int i = 1; i <= N; i++) {
		// 해당 날에 잡힌 상담을 한다면
		dp[i + T[i]] = max(dp[i + T[i]], dp[i] + P[i]);

		// 다음 dp를 계산하기 위함
		dp[i + 1] = max(dp[i + 1], dp[i]);
	}

	cout << dp[N+1] << '\n';
}
