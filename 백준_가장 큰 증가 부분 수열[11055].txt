// 백준 11055 : 가장 큰 증가 부분 수열
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int n;
vector<int> v(1001);
int dp[1001];

int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> v[i];
	}

	for (int i = 1; i <= n; i++) {
		int maximum = 0;
		for (int j = 1; j < i; j++) {
			if (v[j] < v[i] && dp[j] > maximum) {
				maximum = dp[j];
			}
		}
		dp[i] = maximum + v[i];
	}

	int ans = 0;
	for (int i = 1; i <= n; i++) {
		if (ans < dp[i]) ans = dp[i];
	}
	cout << ans << '\n';
}
