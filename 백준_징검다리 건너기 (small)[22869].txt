// 백준 22869 : 징검다리 건너기 (small)
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main() {
	int n, k;
	cin >> n >> k;
	vector<int> a(n);
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}

	vector<bool> dp(n);
	dp[0] = true;
	for (int i = 1; i < n; i++) {
		for (int j = 0; j < i; j++) { // j에서 i로 점프
			if (dp[j] == false) continue;

			int tmp = (i - j) * (1 + abs(a[j] - a[i]));
			if (tmp <= k) {
				dp[i] = true;
				break;
			}
		}
	}

	string ans;
	if (dp[n - 1]) ans = "YES";
	else ans = "NO";

	cout << ans << '\n';
}
