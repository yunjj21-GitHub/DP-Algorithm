// 백준 1463 : 1로 만들기
#include <vector>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;

	int *dp = new int[n + 1]; // 사용할 인덱스 : 1~n

	// 1~3을 만드는 최소 연산의 횟수를 저장
	dp[1] = 0;
	dp[2] = 1; // 1*2 또는 1+1
	dp[3] = 1; // 1*3

	// 4~n을 만드는 최소 연산의 횟수를 저장
	for (int i = 4; i <= n; i++) {
		dp[i] = dp[i - 1] + 1;
		if (i % 2 == 0) dp[i] = min(dp[i], dp[i / 2] + 1); // i / 2가 자연수 라면
		if (i % 3 == 0) dp[i] = min(dp[i], dp[i / 3] + 1); // i / 3이 자연수 라면
	}
	cout << dp[n] << "\n";
}
