// 백준 1912 : 연속합
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	vector<int> num(n);
	for (int i = 0; i < n; i++) {
		cin >> num[i];
	}

	int* dp = new int[n];
	dp[0] = num[0];
	for (int i = 1; i < n; i++) {
		if (num[i] + dp[i - 1] > num[i]) {
			dp[i] = num[i] + dp[i - 1];
		}
		else {
			dp[i] = num[i];
		}
	}

	int ans = -1000;
	for (int i = 0; i < n; i++) {
		if (dp[i] > ans) ans = dp[i];
	}
	cout << ans << '\n';
}
