// 백준 10870 : 피보나치 수 5
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int solution(int n) {
	int *dp = new int[n + 1]; // 사용인덱스 : 0 ~ n

	// 피보나치수는 항상 0과 1로 시작한다.
	dp[0] = 0; // 0번째 피보나치 수
	dp[1] = 1; // 1번째 피보나치 수

	// 2번째 피보나치 수 ~ n번째 피보나치 수 까지를 구함
	for (int i = 2; i <= n; i++) {
		dp[i] = dp[i - 2] + dp[i - 1];
	}
	return dp[n]; // n번째 피보나치수를 반환
}

int main() {
	int input;
	cin >> input;
	cout << solution(input) << endl;
}
