// 백준 11726 : 2Xn 타일링
#include <vector>
#include <string>
#include <iostream>

#define MODULER 10007

using namespace std;

int main() {
	int n;
	cin >> n;

	int* dp = new int[n + 1];

	dp[1] = 1;
	dp[2] = 2;
	
	for (int i = 3; i <= n; i++) {
		dp[i] = dp[i - 1] + dp[i - 2];
		dp[i] %= MODULER;
	}

	cout << dp[n] << '\n';
}
