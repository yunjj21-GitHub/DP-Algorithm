// 백준 2839 : 설탕 배달
#include <vector>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;

int dp[5001]; // 사용할 인덱스 : 1 ~ n

int main() {
	int n; // 상근이가 배달해야 하는 설탕의 무게
	cin >> n;

	dp[3] = dp[5] = 1;

	for (int i = 6; i <= n; i++) {
		// dp[i - 3] !=0 : i-3kg을 만드는 방법이 존재함을 의미
		// dp[i - 5] !=0 : i-5kg을 만드는 방법이 존재함을 의미
		if (dp[i - 3] != 0 && dp[i - 5] !=0) dp[i] = min(dp[i - 3], dp[i - 5]) + 1;
		else if (dp[i - 3] != 0 && dp[i - 5] == 0) dp[i] = dp[i - 3] + 1;
		else if (dp[i - 3] == 0 && dp[i - 5] != 0) dp[i] = dp[i - 5] + 1;
	}

	if (dp[n] == 0) cout << -1 << endl;
	else cout << dp[n] << endl;
}
