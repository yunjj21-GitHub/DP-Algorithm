// 백준 9095 : 1, 2, 3 더하기
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main() {
	int T;
	cin >> T;
	vector<int> answer;
	while (T--) {
		int N;
		cin >> N;

		int* dp = new int[N + 1]; // 사용할 인덱스 : 1 ~ N

		dp[1] = 1;
		dp[2] = 2;
		dp[3] = 4;

		for (int i = 4; i <= N; i++) {
			dp[i] = dp[i - 3] + dp[i - 2] + dp[i - 1];
		}

		answer.push_back(dp[N]);
	}

	for (int ans : answer) {
		cout << ans << '\n';
	}
}
